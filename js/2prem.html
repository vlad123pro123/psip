function saveToLocalStorage() { var nameValue = document.getElementById("nameInput").value; var emailValue = document.getElementById("emailInput").value; var messageValue = document.getElementById("mes").value; localStorage.setItem("name", nameValue); localStorage.setItem("email", emailValue); localStorage.setItem("message", messageValue); } function showAlert() { var nameValue = localStorage.getItem("name") || "Нет данных"; var emailValue = localStorage.getItem("email") || "Нет данных"; var messageValue = localStorage.getItem("message") || "Нет данных"; alert("Имя: " + nameValue + "\nEmail: " + emailValue + "\nСообщение: " + messageValue); } function submitForm() { saveToLocalStorage(); showAlert(); } window.onload = function() { document.getElementById("nameInput").value = localStorage.getItem("name") || ""; document.getElementById("emailInput").value = localStorage.getItem("email") || ""; document.getElementById("mes").value = localStorage.getItem("message") || ""; };
